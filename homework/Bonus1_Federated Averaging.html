class Worker:
    def __init__(self, x, y):
        self.x = x #s-by-d local feature matrix
        self.y = y #s-by-one local label matrix
        self.s = x.shape[0] #number of local samples
        self.d = x.shape[1] #number of features
        self.w = numpy.zeros((d,1)) #d-by-one model parameter vector
    
    #Set the model parameters to the latest
    def set_param(self,w):
        self.w = w
        
    #Compute the local loss
    def loss(self):
        yx = numpy.multiply(self.y, self.x) #s-by-d matrix
        yxw = numpy.dot(yx, self.w) #s-by-one matrix
        vec1 = numpy.exp(-yxw) #s-by-one matrix
        vec2 = numpy.log(1+vec1)
        return numpy.sum(vec2) #loss function
    
    #Compute the local gradient
    def gradient(self):
        yx = numpy.multiply(self.y, self.x)
        yxw = numpy.dot(yx, self.w)
        vec1 = numpy.exp(yxw)
        vec2 = numpy.divide(yx, 1+vec1)
        g = -numpy.sum(vec2, axis=0).reshape(self.d, 1)
        return g
    
    def gd(self, alpha):
        g = self.gradient()
        self.w -= alpha * g
        return self.w
    
    def stochastic_objective_gradient(self, w, xi, yi, lam):
        d = xi.shape[0]
        yx = yi * xi # 1-by-d matrix
        yxw = float(numpy.dot(yx, w)) # scalar

        # calculate stochastic gradient
        g_loss = -yx.T / (1 + numpy.exp(yxw)) # d-by-1 matrix
        g = g_loss + lam * w # d-by-1 matrix

        return g
    
    def sgd(self, stepsize, lam):
        n = self.x.shape[0]

        # randomly shuffle the samples
        rand_indices = numpy.random.permutation(n)
        x_rand = self.x[rand_indices, :]
        y_rand = self.y[rand_indices, :]


        for i in range(n):
            xi = x_rand[i, :] # 1-by-d matrix
            yi = float(y_rand[i, :]) # scalar
            g = self.stochastic_objective_gradient(self.w, xi, yi, lam)
            self.w -= stepsize * g

        stepsize *= 0.9 # decrease step size
            
        return self.w

class Server:
    def __init__(self, m, n, d):
        self.m = m #number of woker nodes
        self.n = n #number of training samples
        self.d = d #number of features
        self.w = numpy.zeros((d, 1)) #d-by-one model parameter vector
        self.g = numpy.zeros((d, 1)) #d-by-one gradient
        self.v = numpy.zeros((d, 1)) #d-by-one momentum
        self.loss = 0 #loss funtion value
        self.obj = 0 #objective function value
        
    def broadcast(self):
        return self.w
    
    #Sum up the gradients and loss function evaluated by the workers
    #grads: a list of d-by-one vectors
    #losses: a list of scalars
    def aggregate(self, grads, losses):
        self.g = numpy.zeros((self.d, 1))
        self.loss = 0
        for k in range (self.m):
            self.g += grads[k]
            self.loss += losses[k]
            
    #Compute the gradient
    def gradient(self, lam):
        self.g = self.g/self.n + lam * self.w
        
    #Compute the objective function
    def objective(self, lam):
        reg = lam/2 * numpy.sum(self.w * self.w)
        self.obj = self.loss/self.n + reg
        return self.obj
    #Update the model parameters using gradient descent
    #alpha: learning rate (step size)
    
    def agd(self, alpha, beta): 
        self.v *= beta
        self.v += self.g
        self.w -= alpha * self.v

        indices = list(range(i*s,(i+1)*s))

def parallel_cal(q, optimization):
    m = 4 #number of worker nodes
    server, workers = create_server_workers(m, x_train, y_train)
    lam = 1E-5 #regularization parameter
    alpha = 1E-2 #learning rate
    beta = 0.9 #momentum parameter
    worker_stepsize = 3E-2
    max_epoch = 300 #number of ephochs
    obj_values =[]
    
    for t in range(max_epoch):
        #step 1: broadcast
        w = server.broadcast()
        for i in range(m):
            workers[i].set_param(w)
            
        #step 2: wokers' local computation
        grads = []
        losses = []
        
        for i in range(m):
            for j in range(q):
                
                if optimization =='GD':
                    w_t = workers[i].gd(worker_stepsize)
                else:
                    w_t = workers[i].sgd(worker_stepsize, lam)
                    
            g = w - w_t
            grads.append(g)
            l = workers[i].loss()
            losses.append(l)   
            
        # step 3: aggregate the workers ’ outputs
        server.aggregate(grads, losses)
        
        # step 4: server update the model parameters
        server.gradient(lam) # compute gradient
        obj = server.objective(lam) # compute objective function 
        obj_values.append(obj)
        print('Objective function value = ' + str(obj))
        server.agd(alpha, beta) # updates the model parameters
        
    return obj_values
